按产品分组聚合

import pandas as pd

df = pd.read_csv('sales.csv')

# 按产品分组，计算销售总额和平均销售额
grouped_df = df.groupby('产品').agg({
    '销售额': ['sum', 'mean'],
    '数量': 'sum'
})

print(grouped_df)
